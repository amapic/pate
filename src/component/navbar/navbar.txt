export function Navbar(props: any) {
  const { scrollY } = useViewportScroll();
  const controls = useAnimation();

  useEffect(() => {
    scrollY.onChange(v => {

      if (v > 0) {
        controls.start("pasenhaut")
      }
      if (v === 0) {
        controls.start("enhaut")
      }
    });

  }, []);


  return (
    <motion.div
      key={shortid.generate()}
      initial="enhaut"
      animate={controls}
      variants={{
        enhautPlanet: {
          height: '80px',
          position: 'sticky',
          top: '0px',
          zIndex: 7
        },
        enhaut: {
          height: '40px',
          position: 'sticky',
          top: '0px',
          zIndex: 7
        },
        pasenhaut: {
          height: '40px',
          position: 'sticky',
          top: '0px',
          zIndex: 7
        }
      }}
      transition={{ type: "spring", duration: 1 }}

    >
      <nav className="">

        {props.children}
      </nav>
    </motion.div>

  );
}
